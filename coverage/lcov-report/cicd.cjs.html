function checkCICDSetup(projectDir) {
  const checks = [];
 
  // Check GitHub workflow
  const workflowFile = path.join(projectDir, '.github', 'workflows', 'contract-validation.yml');
  checks.push({
    name: 'GitHub Actions workflow',
    exists: fs.existsSync(workflowFile),
    path: '.github/workflows/contract-validation.yml',
  });
 
  // Check PR template
  const prTemplate = path.join(projectDir, '.github', 'PULL_REQUEST_TEMPLATE.md');
  checks.push({
    name: 'PR template',
    exists: fs.existsSync(prTemplate),
    path: '.github/PULL_REQUEST_TEMPLATE.md',
  });
 
  // Check pre-commit hook
  const preCommit = path.join(projectDir, '.git', 'hooks', 'pre-commit');
  checks.push({
    name: 'Pre-commit hook',
    exists: fs.existsSync(preCommit),
    path: '.git/hooks/pre-commit',
  });
 
  // Check grabby config
  const grabbyConfig = path.join(projectDir, '.grabby', 'config.yaml');
  checks.push({
    name: 'Grabby config',
    exists: fs.existsSync(grabbyConfig),
    path: '.grabby/config.yaml',
  });
 
  return {
    checks,
    allConfigured: checks.every(c => c.exists),
    missing: checks.filter(c => !c.exists),
  };
}
 
/**
 * Format CI/CD setup report
 */
function formatSetupReport(setup) {
  const lines = [];
 
  lines.push('CI/CD Setup Status');
  lines.push('─'.repeat(40));
 
  for (const check of setup.checks) {
    const icon = check.exists ? '✓' : '✗';
    const color = check.exists ? '\x1b[32m' : '\x1b[31m';
    lines.push(`${color}${icon}\x1b[0m ${check.name}`);
    if (!check.exists) {
      lines.push(`  └─ Missing: ${check.path}`);
    }
  }
 
  lines.push('');
  if (setup.allConfigured) {
    lines.push('\x1b[32m✓ CI/CD fully configured\x1b[0m');
  } else {
    lines.push(`\x1b[33m! ${setup.missing.length} item(s) not configured\x1b[0m`);
    lines.push('Run: grabby cicd --setup');
  }
 
  return lines.join('\n');
}
